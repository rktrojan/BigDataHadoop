##Sqoop commands for Hadoop VM

#EXAMPLE 1  -- make sure that target folder is not already created
sqoop import \
--connect jdbc:mysql://localhost:3306/retail_db \
--username root \
--password cloudera \
--table orders \
--target-dir /user/cloudera/orders 


#EXAMPLE 2
sqoop import \
--connect jdbc:mysql://localhost:3306/retail_db \
--username root \
--password cloudera \
--table orders \
--target-dir /user/cloudera/orders \
-m 10 \
--split-by oid


#EXAMPLE 3
#instead of a table if we want run a query and import the query output
sqoop import \
--connect jdbc:mysql://localhost:3306/retail_db \
--username root \
--password cloudera \
--query 'select * from orders where $CONDITIONS' \
--target-dir /user/cloudera/orders \
-m 10 \
--split-by oid
